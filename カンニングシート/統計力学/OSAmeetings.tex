\section*{\Large{各種公式}}
\subsection*{分配関数との関係式}
\begin{eqnarray}
  &&F = -\frac{1}{\beta}\ln Z\\
  &&E = -\frac{\partial}{\partial\beta}\ln Z
\end{eqnarray}
\subsection*{熱力学的関係式}
\begin{eqnarray}
  F = E-TS,\quad dF = -pdV -SdT 
\end{eqnarray}
\subsection*{磁場との関係式}
\begin{eqnarray}
  H = \cdots - \mu_0H\sum_i \sigma_i
\end{eqnarray}
のとき
\begin{eqnarray}
  &&m= -\frac{\partial F}{\partial H}\\
  &&\chi = \frac{\partial m}{\partial H}|_{H=0}
\end{eqnarray}
